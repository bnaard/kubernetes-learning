% --- PAGE 3: SERVICES & NETWORKING ---
\begin{center}\textbf{\Large CKA CHEATSHEET: PAGE 3}\\\textit{Services \& Networking}\end{center}

\sectionheader{1. Service Types}
\begin{itemize}
    \item \textbf{ClusterIP}: Internal (Default)
    \item \textbf{NodePort}: Static port on each node (30000-32767)
    \item \textbf{LoadBalancer}: Cloud-provider external IP
\end{itemize}

\sectionheader{2. Ingress Rules}
% [CLI: Insert TikZ for Ingress: Path /blue -> svc-blue]
\begin{lstlisting}
spec:
  rules:
  - host: myapp.com
    http:
      paths:
      - path: /blue
        pathType: Prefix
        backend:
          service:
            name: blue-svc
            port: {number: 80}
\end{lstlisting}

\sectionheader{3. Network Policies}
\begin{lstlisting}
# Isolation is key. No rule = Deny all.
podSelector: {matchLabels: {app: db}}
ingress:
- from:
  - podSelector: {matchLabels: {app: web}}
\end{lstlisting}

\sectionheader{1. RBAC Workflow}
\begin{lstlisting}
# 1. Create Role (Permissions)
k create role pod-reader --verb=get,list --resource=pods
# 2. Create RoleBinding (Link to user/SA)
k create rolebinding read-pods --role=pod-reader --serviceaccount=default:user-sa
\end{lstlisting}

\sectionheader{2. Service Accounts}
\begin{lstlisting}

# Assign to Pod:
spec:
  serviceAccountName: developer-sa
\end{lstlisting}

\sectionheader{1. The Storage Handshake}
% [CLI: Insert TikZ for PV -> PVC -> Pod Mount]
\begin{lstlisting}
# PV (Cluster-wide)
accessModes: [ReadWriteOnce]
hostPath: {path: "/data"}

# PVC (Namespace-scoped)
resources:
  requests: {storage: 1Gi}
\end{lstlisting}

\sectionheader{2. Volume Mounts}

  volumes:
  - name: my-volume
    persistentVolumeClaim:
      claimName: my-pvc
\end{lstlisting}
