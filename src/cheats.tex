% === Shell Aliases & Env Vars =============================================
\tcbsubtitle{Shell Aliases}
\begin{cheatcode}
alias k='kubectl'
alias kgp='kubectl get pods'
alias kgs='kubectl get svc'
alias kgn='kubectl get nodes'
export do='--dry-run=client -o yaml'
export now='--force --grace-period=0'
source <(kubectl completion bash)
complete -F __start_kubectl k
\end{cheatcode}

% === Context & Namespace ==================================================
\tcbsubtitle{Context \& Namespace}
\begin{cheatcode}
k config get-contexts
k config use-context <ctx-name>
k config set-context --current --namespace=<ns>
k get all -n <ns>
k get all -A          # all namespaces
k api-resources --namespaced=true
\end{cheatcode}

% === Vim: YAML Editing ====================================================
\tcbsubtitle{Vim: YAML Editing}
\begin{cheatcode}
# In Vim (run after opening file):
:set ts=2 sw=2 et    # 2-space YAML indent
# Visual block indent:
Ctrl+V  select lines  >  (right) or  <  (left)
# Mass prepend # comment:
Ctrl+V  select lines  Shift+I  #  Esc
gg=G                 # re-indent entire file
\end{cheatcode}

% === Imperative Templates =================================================
\tcbsubtitle{Imperative Templates}
\begin{cheatcode}
k run <name> --image=<img> $do > pod.yaml
k create deploy <name> --image=<img> \
  --replicas=3 $do > deploy.yaml
k expose deploy <name> --port=80 \
  --type=ClusterIP $do
k create job <name> --image=<img> $do
k create cj <name> --image=<img> \
  --schedule="*/5 * * * *" $do
k create cm <name> --from-literal=k=v
k create secret generic <name> \
  --from-literal=pwd=s3cr3t
k create sa <name> -n <ns>
k create role r --verb=get,list \
  --resource=pods
k create rolebinding rb --role=r \
  --serviceaccount=<ns>:<sa>
\end{cheatcode}

% === Useful kubectl ========================================================
\tcbsubtitle{Useful kubectl}
\begin{cheatcode}
k explain pod.spec.containers
k describe pod <name>     # check Events!
k exec -it <pod> -- bash
k logs <pod> [-c <container>] [--previous]
k port-forward pod/<name> 8080:80
k cp <pod>:/path /local
k get events --sort-by='.lastTimestamp' -A
k delete pod <name> $now  # force delete
k label pod <name> key=val
k annotate pod <name> note="txt"
\end{cheatcode}
