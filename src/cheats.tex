% === Shell Aliases & Env Vars =============================================
\tcbsubtitle{Shell Aliases}
\begin{cheatcodebashbold}
alias k='kubectl'
export do='--dry-run=client -o yaml'
\end{cheatcodebashbold}

\begin{cheatcodebash}
alias kgp='kubectl get pods'
alias kgs='kubectl get svc'
alias kgn='kubectl get nodes'
export now='--force --grace-period=0'
source <(kubectl completion bash)
complete -F __start_kubectl k
\end{cheatcodebash}

% === Context & Namespace ==================================================
\tcbsubtitle{Context \& Namespace}
\begin{cheatcodebashbold}
k config set-context --current --namespace=<ns>
k get all -A          # all namespaces
\end{cheatcodebashbold}

\begin{cheatcodebash}
k config get-contexts
k config use-context <ctx-name>
k get all -n <ns>
k api-resources --namespaced=true
\end{cheatcodebash}

% === Imperative Templates =================================================
\tcbsubtitle{Imperative Templates}
\begin{cheatcodebashbold}
### Run container, tee prints on tty and into file
k run <name> --image=<img> $do | tee pod.yaml
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
### Define deployment with 3 replicas
k create deploy <name> --image=<img> --replicas=3 $do > d.yaml
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
### Expose deployment as service cluster internally
k expose deploy <name> --port=80 --type=ClusterIP $do
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
### Create job / cronjob
k create job <name> --image=<img> $do
k create cj <name> --image=<img> --schedule="*/5 * * * *" $do
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
### Create configmap / secret / service account
k create cm <name> --from-literal=k=v
k create cm <name> --from-file=<filepath> # pairs in file
k create cm <name> --from-env-file=<filepath>
k create secret generic <name> --from-literal=pwd=s3cr3t
k create sa <name> -n <ns>
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
### Create role and bind it
k create role r --verb=get,list --resource=pods
k create rolebinding rb --role=r --serviceaccount=<ns>:<sa>
\end{cheatcodebashbold}

% === Useful kubectl ========================================================
\tcbsubtitle{Useful kubectl}
\begin{cheatcodebashbold}
### Inspect / debug
k describe pod <name>              # check Events!
k logs <pod> [-c <container>] [-p] # check logs [previous] 
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
### tunnel from port 8080 on localhost to port 80 of pod
### CKA: verify pod's web UI/endpoint is working
k port-forward pod/<name> 8080:80
\end{cheatcodebashbold}

\begin{cheatcodebash}
### Explore API fields
k explain pod.spec.containers
\end{cheatcodebash}

\begin{cheatcodebash}
### Files / events / lifecycle
k cp <pod>:/path /local
k get events --sort-by='.lastTimestamp' -A
k delete pod <name> $now       # force delete
\end{cheatcodebash}

\begin{cheatcodebash}
### Metadata
k label pod <name> key=val
k annotate pod <name> note="txt"
\end{cheatcodebash}
