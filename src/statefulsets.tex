% === StatefulSets ==========================================================
\tcbsubtitle{Overview}
{\fontsize{6.5}{7.5}\selectfont\sffamily
Like a Deployment but for \textbf{stateful} workloads. Each Pod gets a \textbf{stable identity} (ordinal index), a \textbf{persistent hostname}, and its own \textbf{PVC} that survives rescheduling. Pods are created/deleted \textbf{in order} (0, 1, 2\ldots).\par}

\tcbsubtitle{Imperative}
\cmdhead{Exam shortcut: scaffold from Deployment}
\begin{cheatcodebash}
k create deploy db --image=mysql:8 --replicas=3 $do > sts.yaml
# Then edit sts.yaml:
#  1. kind: Deployment → StatefulSet
#  2. Add spec.serviceName: <headless-svc>
#  3. Add spec.volumeClaimTemplates (if needed)
#  4. Remove spec.strategy (uses updateStrategy instead)
\end{cheatcodebash}

\cmdhead{Create and scale}
\begin{cheatcodebash}
# No direct "k create statefulset" — use YAML
k get statefulsets      # or: k get sts
k scale sts <name> --replicas=5
k delete sts <name>                # keeps PVCs!
k delete sts <name> --cascade=orphan  # keeps Pods too
\end{cheatcodebash}

\tcbsubtitle{Spec Essentials}
\begin{cheatcodebashbold}
apiVersion: apps/v1
kind: StatefulSet
metadata: {name: db}
spec:
  serviceName: db-headless   # required! headless Service
  replicas: 3
  selector: {matchLabels: {app: db}}
  template:
    metadata: {labels: {app: db}}
    spec:
      containers:
      - name: mysql
        image: mysql:8
        ports: [{containerPort: 3306}]
        volumeMounts:
        - {name: data, mountPath: /var/lib/mysql}
  volumeClaimTemplates:      # one PVC per Pod
  - metadata: {name: data}
    spec:
      accessModes: ["ReadWriteOnce"]
      resources:
        requests: {storage: 10Gi}
\end{cheatcodebashbold}

\tcbsubtitle{Headless Service (required)}
\begin{cheatcodebash}
apiVersion: v1
kind: Service
metadata: {name: db-headless}
spec:
  clusterIP: None            # headless!
  selector: {app: db}
  ports:
  - {port: 3306, targetPort: 3306}
\end{cheatcodebash}

  Pod names & Random hash & Ordinal index \\
  Storage & Shared PVC & PVC per Pod \\
  Startup order & Parallel & Sequential \\

\tcbsubtitle{Help}
\cmdhead{kubectl}
\begin{cheatcodebash}
k explain statefulset.spec --recursive
k explain statefulset.spec.volumeClaimTemplates
\end{cheatcodebash}

\cmdhead{\url{https://kubernetes.io/docs/}}
\begin{cheatcodebash}
search: "statefulsets"
\end{cheatcodebash}

\warning{Deleting a StatefulSet does \textbf{not} delete its PVCs. Clean up manually: \texttt{k delete pvc data-db-\{0,1,2\}}}
