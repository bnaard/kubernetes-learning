% === Network Policies =====================================================
\tcbsubtitle{Network Policies}
\begin{cheatcodebash}
# Default: all traffic allowed. Once a NetworkPolicy
# selects a pod: only explicitly allowed traffic passes
k get netpol -A
k describe netpol <name>
\end{cheatcodebash}

\begin{cheatcodebash}
# Deny all ingress + egress to a pod:
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata: {name: deny-all, namespace: <ns>}
spec:
  podSelector: {}          # matches all pods in ns
  policyTypes: [Ingress, Egress]
\end{cheatcodebash}

  \begin{cheatcodebash}
# Allow ingress only from specific pods:
spec:                  # <networkpolicy>.spec
  podSelector: {matchLabels: {app: db}}
  policyTypes: [Ingress]
  ingress:
  - from:
    - podSelector:          # AND condition within {}
        matchLabels: {app: web}
    - namespaceSelector:    # OR: separate list item
        matchLabels: 
          env: prod         # namespace's labels
          kubernetes.io/metadata.name: production # auto!
    ports:
    - {protocol: TCP, port: 5432} # ex. PostgreSQL port
# Allow egress to DNS only:
  egress:
  - ports:
    - {port: 53, protocol: UDP}
    - {port: 53, protocol: TCP}
\end{cheatcodebash}

\begin{cheatcodebash}
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-ingress
  annotations:
    # client → myapp.example.com/api →  api-svc receives /
    nginx.ingress.kubernetes.io/rewrite-target: /
spec:
  ingressClassName: nginx
  rules:
  - host: myapp.example.com
    http:
      paths:
      - path: /api  # rewritten to /
        pathType: Prefix # Prefix|Exact|ImplementationSpecific
        backend:
          service:
            name: api-svc
            port: {number: 80}
  tls:
  - hosts: [myapp.example.com]
    secretName: tls-secret
\end{cheatcodebash}

% === Gateway API ===========================================================
\tcbsubtitle{Gateway API (successor to Ingress)}
\begin{cheatcodebash}
apiVersion: gateway.networking.k8s.io/v1
kind: HTTPRoute
metadata: {name: my-route}
spec:
  parentRefs:
    - name: my-gateway         # references a Gateway object
  hostnames: ["myapp.example.com"]
  rules:
    - matches:
        - path: {type: PathPrefix, value: /api}
      filters:
        - type: URLRewrite
          urlRewrite:
            path:
              type: ReplacePrefixMatch
              replacePrefixMatch: / # /api/users -> /users
      backendRefs:
        - {name: api-svc, port: 80}
    - matches:
        - headers:
            - {name: X-Env, value: canary}
      backendRefs:
        - {name: api-svc-v2, port: 80, weight: 100}
\end{cheatcodebash}
