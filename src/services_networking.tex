        \tcbsubtitle{Service Types}

          \begin{tcolorbox}
              [
              skin=cheatboxtablesimple,
              before skip=0pt,
              after skip=0pt,
              tabularray={
              colspec={X[0.5,l]X[2,l]},
              columns={valign=t, colsep=0.35em},
              rows={rowsep=0pt, font=\sffamily\footnotesize},   % \fontsize{5}{6}\selectfont
              hline{2}={solid, wd=0.3pt, fg=gray!40},
              vline{2}={dash=solid, wd=0.3pt, fg=gray!30},
              row{1}={font=\sffamily\bfseries\footnotesize},  % \fontsize{5}{6}\selectfont
              }
              ]
              \textbf{Type}    & \textbf{Description}         \\
              ClusterIP        & Internal only (default). Gets a virtual cluster IP reachable only from within the cluster.      \\
              Headless         & \texttt{clusterIP: None} — no virtual IP. DNS returns Pod IPs directly, bypasses kube-proxy. Use: StatefulSets, service discovery without load balancing \\
              NodePort         & Everything ClusterIP does \textbf{+} opens a static port (30000--32767) on every node's real IP for external access.      \\
              LoadBalancer     & Everything NodePort does \textbf{+} provisions a cloud LB with an external IP.      \\
              ExternalName     & DNS CNAME alias to external host (no ClusterIP, no proxying).      \\

          \end{tcolorbox}
          \vspace{0.5ex}

        \cmdhead{Imperative}
        \begin{cheatcodebash}
k expose deploy <name> --port=80 --target-port=8080 --type=ClusterIP
k expose deploy <name> --port=80 --target-port=8080 --type=NodePort
k create svc nodeport <name> --tcp=80:8080 --node-port=30080
k expose deploy <name> --port=80 --type=LoadBalancer
# Headless: ClusterIP service with clusterIP=None
k create svc clusterip <name> --clusterip="None" --tcp=80:8080
k get svc
k get endpoints <svc>    # empty = label mismatch!
        \end{cheatcodebash}

        % === Ingress ==============================================================
        \tcbsubtitle{Ingress}
        \cmdhead{Imperative}
        \begin{cheatcodebash}
        k get ingressclass
        \end{cheatcodebash}

        \cmdhead{Spec Essentials}
        \begin{cheatcodebash}
        apiVersion: networking.k8s.io/v1
        kind: Ingress
        metadata:
          name: my-ingress
          annotations:
            # client → myapp.example.com/api →  api-svc receives /
            nginx.ingress.kubernetes.io/rewrite-target: /
        spec:
          ingressClassName: nginx
          rules:
          - host: myapp.example.com
            http:
              paths:
              - path: /api  # rewritten to /
                pathType: Prefix # Prefix|Exact|ImplementationSpecific
                backend:
                  service:
                    name: api-svc
                    port: {number: 80}
          tls:
          - hosts: [myapp.example.com]
            secretName: tls-secret
        \end{cheatcodebash}

        \cmdhead{Helpers}
        \begin{cheatcodebash}
        k explain ingress.spec --recursive
        \end{cheatcodebash}

        \cmdhead{\url{https://kubernetes.io/docs/}}

        % === Network Policies =====================================================
        \tcbsubtitle{Network Policies}
    
        \cmdhead{Imperative}
        \begin{cheatcodebash}
        # Default: all traffic allowed. Once a NetworkPolicy
        # selects a pod: only explicitly allowed traffic passes
        k get netpol -A
        k describe netpol <name>
        \end{cheatcodebash}

        \cmdhead{Spec Essentials: Deny all ingress + egress to a pod}
        \begin{cheatcodebash}        
        apiVersion: networking.k8s.io/v1
        kind: NetworkPolicy
        metadata: {name: deny-all, namespace: <ns>}
        spec: 
          podSelector: {}        # matches all pods in ns
          policyTypes: [Ingress, Egress]
        \end{cheatcodebash}

        \cmdhead{Spec Essentials: Allow ingress only from specific pods}
        \begin{cheatcodebash}
        spec:                    # <networkpolicy>.spec
          podSelector: {matchLabels: {app: db}}
          policyTypes: [Ingress, Egress] # directions this policy governs
          ingress:                       # (unlisted = not touched)
          - from:
            - podSelector:       # AND condition within {}
                matchLabels: {app: web}
            - namespaceSelector: # OR: separate list item
                matchLabels:
                  env: prod      # namespace's labels
                  kubernetes.io/metadata.name: production # auto!
            ports:
            - {protocol: TCP, port: 5432} # ex. PostgreSQL port
          egress:  # Allow egress to DNS only
          - ports:
            - {port: 53, protocol: UDP}
            - {port: 53, protocol: TCP}
        \end{cheatcodebash}

        \cmdhead{Helpers}
        \begin{cheatcodebash}
        k explain networkpolicy.spec --recursive
        \end{cheatcodebash}

        \cmdhead{\url{https://kubernetes.io/docs/}}
        \begin{cheatcodebash}
        search: "network policies"
        \end{cheatcodebash}

        \tcbsubtitle{CoreDNS}
        \cmdhead{DNS Record Patterns}
        \begin{cheatcodebash}
<svc>.<ns>.svc.cluster.local        # Service DNS
<ip-dashes>.<ns>.pod.cluster.local  # Pod by IP
# Pod with hostname + subdomain fields set:
<hostname>.<subdomain>.<ns>.svc.cluster.local
# Requires pod spec: hostname: myhost, subdomain: mysvc
# + a matching headless Service named "mysvc"
# StatefulSet pods (via headless Service):
<sts-name>-<ordinal>.<headless-svc>.<ns>.svc.cluster.local
        \end{cheatcodebash}

        \cmdhead{Edit CoreDNS config \& apply}
        \begin{cheatcodebash}
        k get cm coredns -n kube-system -o yaml
        k edit cm coredns -n kube-system
        # After editing, restart to pick up changes:
        k rollout restart deploy coredns -n kube-system
        k rollout status  deploy coredns -n kube-system
        \end{cheatcodebash}

        % === Gateway API ===========================================================
        \tcbsubtitle{Gateway API (successor to Ingress)}
        \cmdhead{Gateway Object (Cilium example)}
        \begin{cheatcodebash}
        apiVersion: gateway.networking.k8s.io/v1
        kind: Gateway
        metadata: {name: my-gateway}
        spec:
          gatewayClassName: cilium      # provided by Cilium
          listeners:
          - name: http
            protocol: HTTP
            port: 80
            allowedRoutes:
              namespaces: {from: Same}
        \end{cheatcodebash}

        \cmdhead{Gateway with HTTPS / TLS termination}
        \begin{cheatcodebash}
        apiVersion: gateway.networking.k8s.io/v1
        kind: Gateway
        metadata: {name: my-tls-gateway}
        spec:
          gatewayClassName: cilium
          listeners:
          - name: https
            protocol: HTTPS
            port: 443
            tls:
              mode: Terminate            # terminate TLS at gateway
              certificateRefs:
              - name: my-tls-secret      # existing Secret (type: tls)
            allowedRoutes:
              namespaces: {from: Same}
        \end{cheatcodebash}
        {\fontsize{4.5}{5.5}\selectfont\sffamily\color{gray}%
        Secret must be \texttt{type: kubernetes.io/tls} with \texttt{tls.crt} and \texttt{tls.key} keys.
        Create with: \texttt{k create secret tls my-tls-secret --cert=tls.crt --key=tls.key}\par}

        \cmdhead{HTTPRoute Spec}
        \begin{cheatcodebash}
        apiVersion: gateway.networking.k8s.io/v1
        kind: HTTPRoute
        metadata: {name: my-route}
        spec:
          parentRefs:
            - name: my-gateway         # references a Gateway object
          hostnames: ["myapp.example.com"]
          rules:
            - matches:
                - path: {type: PathPrefix, value: /api}
              filters:
                - type: URLRewrite
                  urlRewrite:
                    path:
                      type: ReplacePrefixMatch
                      replacePrefixMatch: / # /api/users -> /users
              backendRefs:
                - {name: api-svc, port: 80}
            - matches:               # header-based routing
                - headers:
                  - name: X-Env
                    value: staging   # Exact match (default)
              backendRefs:
                - {name: staging-svc, port: 80}
            - matches:               # weighted traffic split
                - path: {type: PathPrefix, value: /}
              backendRefs:           # 80/20 canary split
                - name: api-svc
                  port: 80
                  weight: 80         # 80% -> stable
                - name: api-svc-v2
                  port: 80
                  weight: 20         # 20% -> canary
        \end{cheatcodebash}
