% === Service Types =========================================================
\tcbsubtitle{Service Types}
{\small\sffamily
\begin{tblr}{
  colspec={lX},
  hline{2}={0.4pt},
  rows={rowsep=0.5pt},
  row{1}={font=\bfseries},
}
Type & Description \\
ClusterIP & Internal only (default) \\
NodePort & Via \texttt{NodeIP:Port} (30000--32767) \\
LoadBalancer & Cloud LB, external IP \\
ExternalName & DNS CNAME alias to external host \\
\end{tblr}}
\begin{cheatcode}
k expose deploy <name> --port=80 \
  --target-port=8080 --type=ClusterIP
k expose deploy <name> --port=80 \
  --type=NodePort --node-port=30080
k get svc
k get endpoints <svc>    # empty = label mismatch!
\end{cheatcode}

Service YAML:
\begin{cheatcode}
apiVersion: v1
kind: Service
spec:
  type: NodePort
  selector: {app: web}         # matches pod labels
  ports:
  - port: 80                   # cluster (virtual) port
    targetPort: 8080           # pod's containerPort
    nodePort: 30080            # node port (optional)
\end{cheatcode}

% === Network Policies =====================================================
\tcbsubtitle{Network Policies}
\begin{cheatcode}
# Default: all traffic allowed
# Once a NetworkPolicy selects a pod:
#   only explicitly allowed traffic passes
k get netpol -A
k describe netpol <name>

# Deny all ingress + egress to a pod:
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata: {name: deny-all, namespace: <ns>}
spec:
  podSelector: {}          # matches all pods in ns
  policyTypes: [Ingress, Egress]
---
# Allow ingress only from specific pods:
spec:
  podSelector: {matchLabels: {app: db}}
  policyTypes: [Ingress]
  ingress:
  - from:
    - podSelector:          # AND condition within {}
        matchLabels: {app: web}
    - namespaceSelector:    # OR: separate list item
        matchLabels: {env: prod}
    ports:
    - {protocol: TCP, port: 5432}
# Allow egress to DNS only:
  egress:
  - ports:
    - {port: 53, protocol: UDP}
    - {port: 53, protocol: TCP}
\end{cheatcode}

% === Ingress ==============================================================
\tcbsubtitle{Ingress}
\begin{cheatcode}
k get ingressclass
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: my-ingress
  annotations:
    nginx.ingress.kubernetes.io/rewrite-target: /
spec:
  ingressClassName: nginx
  rules:
  - host: myapp.example.com
    http:
      paths:
      - path: /api
        pathType: Prefix   # Prefix|Exact|ImplementationSpecific
        backend:
          service:
            name: api-svc
            port: {number: 80}
  tls:
  - hosts: [myapp.example.com]
    secretName: tls-secret
\end{cheatcode}

% === Gateway API ===========================================================
\tcbsubtitle{Gateway API (successor to Ingress)}
\begin{cheatcode}
apiVersion: gateway.networking.k8s.io/v1
kind: HTTPRoute
metadata: {name: my-route}
spec:
  parentRefs:
  - name: my-gateway         # references a Gateway object
  hostnames: ["myapp.example.com"]
  rules:
  - matches:
    - path: {type: PathPrefix, value: /api}
    backendRefs:
    - {name: api-svc, port: 80}
  - matches:
    - headers:
      - {name: X-Env, value: canary}
    backendRefs:
    - {name: api-svc-v2, port: 80, weight: 100}
\end{cheatcode}
