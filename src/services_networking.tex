        ClusterIP        & Internal only (default)      \\
        Headless         & \texttt{clusterIP: None} — DNS returns Pod IPs directly, no proxy. Use: StatefulSets, service discovery without load balancing \\

        LoadBalancer     & Cloud LB, external IP      \\
        ExternalName     & DNS CNAME alias to external host      \\

    \cmdhead{Imperative}
    \begin{cheatcodebash}
    k expose deploy <name> --port=80 --target-port=8080 --type=ClusterIP
    k expose deploy <name> --port=80 --target-port=8080 --type=NodePort --node-port=30080
    k expose deploy <name> --port=80 --type=LoadBalancer
    # Headless: ClusterIP service with clusterIP=None
    k create svc clusterip <name> --clusterip="None" --tcp=80:8080
    k get svc
    k get endpoints <svc>    # empty = label mismatch!
    \end{cheatcodebash}

    \cmdhead{Spec Essentials: ClusterIP}
    \begin{cheatcodebash}
    apiVersion: v1
    kind: Service

      type: ClusterIP              # default if omitted
      selector: {app: web}
      ports:
      - port: 80                   # Service (virtual) port

        % === Network Policies =====================================================
        \tcbsubtitle{Network Policies}
    
        \cmdhead{Imperative}
        \begin{cheatcodebash}
        # Default: all traffic allowed. Once a NetworkPolicy
        # selects a pod: only explicitly allowed traffic passes
        k get netpol -A
        k describe netpol <name>
        \end{cheatcodebash}

        \cmdhead{Spec Essentials: Deny all ingress + egress to a pod}
        \begin{cheatcodebash}        
        apiVersion: networking.k8s.io/v1
        kind: NetworkPolicy
        metadata: {name: deny-all, namespace: <ns>}
        spec: 
          podSelector: {}        # matches all pods in ns
          policyTypes: [Ingress, Egress]
        \end{cheatcodebash}

        \cmdhead{Spec Essentials: Allow ingress only from specific pods}
        \begin{cheatcodebash}
        spec:                    # <networkpolicy>.spec
          podSelector: {matchLabels: {app: db}}
          policyTypes: [Ingress]
          ingress:
          - from:
            - podSelector:       # AND condition within {}
                matchLabels: {app: web}
            - namespaceSelector: # OR: separate list item
                matchLabels: 
                  env: prod      # namespace's labels
                  kubernetes.io/metadata.name: production # auto!
            ports:
            - {protocol: TCP, port: 5432} # ex. PostgreSQL port
          egress:  # Allow egress to DNS only
          - ports:
            - {port: 53, protocol: UDP}
            - {port: 53, protocol: TCP}
        \end{cheatcodebash}

        \cmdhead{Help}
        \begin{cheatcodebash}
        k explain networkpolicy.spec --recursive
        \end{cheatcodebash}

        \cmdhead{\url{https://kubernetes.io/docs/}}
        \begin{cheatcodebash}
        service: "network policies"
        \end{cheatcodebash}

        % === Ingress ==============================================================
        \tcbsubtitle{Ingress}
        \cmdhead{Imperative}
        \begin{cheatcodebash}
        k get ingressclass
        \end{cheatcodebash}

        \cmdhead{Spec Essentials}
        \begin{cheatcodebash}
        apiVersion: networking.k8s.io/v1
        kind: Ingress
        metadata:
          name: my-ingress
          annotations:
            # client → myapp.example.com/api →  api-svc receives /
            nginx.ingress.kubernetes.io/rewrite-target: /
        spec:
          ingressClassName: nginx
          rules:
          - host: myapp.example.com
            http:
              paths:
              - path: /api  # rewritten to /
                pathType: Prefix # Prefix|Exact|ImplementationSpecific
                backend:
                  service:
                    name: api-svc
                    port: {number: 80}
          tls:
          - hosts: [myapp.example.com]
            secretName: tls-secret
        \end{cheatcodebash}

        \cmdhead{Help}
        \begin{cheatcodebash}
        k explain ingress.spec --recursive
        \end{cheatcodebash}

        \cmdhead{\url{https://kubernetes.io/docs/}}

        % === Gateway API ===========================================================
        \tcbsubtitle{Gateway API (successor to Ingress)}
        \cmdhead{Gateway Object (Cilium example)}
        \begin{cheatcodebash}
        apiVersion: gateway.networking.k8s.io/v1
        kind: Gateway
        metadata: {name: my-gateway}
        spec:
          gatewayClassName: cilium      # provided by Cilium
          listeners:
          - name: http
            protocol: HTTP
            port: 80
            allowedRoutes:
              namespaces: {from: Same}
        \end{cheatcodebash}

        \cmdhead{HTTPRoute Spec}
        \begin{cheatcodebash}
        apiVersion: gateway.networking.k8s.io/v1
        kind: HTTPRoute
        metadata: {name: my-route}
        spec:
          parentRefs:
            - name: my-gateway         # references a Gateway object
          hostnames: ["myapp.example.com"]
          rules:
            - matches:
                - path: {type: PathPrefix, value: /api}
              filters:
                - type: URLRewrite
                  urlRewrite:
                    path:
                      type: ReplacePrefixMatch
                      replacePrefixMatch: / # /api/users -> /users
              backendRefs:
                - {name: api-svc, port: 80}
            - matches:
                - headers:
                    - {name: X-Env, value: canary}
              backendRefs:
                - {name: api-svc-v2, port: 80, weight: 100}
        \end{cheatcodebash}
