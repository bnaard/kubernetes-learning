% === Deployments ===========================================================
\tcbsubtitle{Imperative}

\cmdhead{Create and scale}
\begin{cheatcodebashbold}
k create deploy <name> --image=<img> --replicas=3
k scale deploy <name> --replicas=5
\end{cheatcodebashbold}

\cmdhead{Define deployment with 3 replicas}
\begin{cheatcodebashbold}
k create deploy <name> --image=<img> --replicas=3 $do > d.yaml
\end{cheatcodebashbold}

\cmdhead{Update image}
\begin{cheatcodebashbold}
k set image deploy/<name> <c>=<img>:tag # container id/name
\end{cheatcodebashbold}

\cmdhead{Rollout management}
\begin{cheatcodebashbold}
k rollout status  deploy/<name>
k rollout history deploy/<name>
k rollout undo    deploy/<name>
k rollout undo    deploy/<name> --to-revision=2
k rollout pause   deploy/<name>
k rollout resume  deploy/<name>
\end{cheatcodebashbold}

\cmdhead{Annotate last section of rollout history}
\begin{cheatcodebashbold}
k annotate deploy/<name> kubernetes.io/change-cause="xy"
\end{cheatcodebashbold}


\tcbsubtitle{Spec Essentials}
\begin{cheatcodebashbold}
spec:                    # <deployment>.spec
  replicas: 3
  selector: {matchLabels: {app: web}} # must match template metadata labels
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1        # extra pods during upd
      maxUnavailable: 0  # min avail. dur upd
  template:
    metadata: {labels: {app: web}} # injected in every pod
    spec: ...            # same as pod spec
\end{cheatcodebashbold}
