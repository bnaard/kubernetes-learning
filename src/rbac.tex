    % === RBAC Overview ========================================================
    \cmdhead{Create Role (namespaced permissions)}
    \begin{cheatcodebash}
    k create role <name> --verb=get,list,watch --resource=pods,deployments -n <ns>
    \end{cheatcodebash}

    \cmdhead{Create ClusterRole (non-namespaced, cluster-wide permissions)}
    \begin{cheatcodebash}
    k create clusterrole <name> --verb='*' --resource='*'
    \end{cheatcodebash}

    \cmdhead{Bind to ServiceAccount}
    \begin{cheatcodebash}
    k create rolebinding <name> --role=<role> --serviceaccount=<ns>:<sa> -n <ns>
    k create clusterrolebinding <name> --clusterrole=<cr> --user=<user>
    \end{cheatcodebash}

    \cmdhead{Verify}
    \begin{cheatcodebash}
    k auth can-i get pods --as=<user>
    k auth can-i get pods --as=system:serviceaccount:<ns>:<sa>
    k auth can-i --list -n <ns>
    \end{cheatcodebash}

    \cmdhead{Spec Essentials}
    \begin{cheatcodebash}
    apiVersion: rbac.authorization.k8s.io/v1
    kind: Role
    metadata: {name: pod-reader, namespace: <ns>}
    rules:
    - apiGroups: [""]           # "" = core API group
      resources: ["pods", "pods/log"]
      verbs: ["get","list","watch"]
    - apiGroups: ["apps"]
      resources: ["deployments"]
      verbs: ["get","list","create","update"]
    \end{cheatcodebash}

      \texttt{get} & Read a single resource by name \\
      \texttt{list} & List all resources of a type \\
      \texttt{watch} & Stream changes (used by controllers) \\
      \texttt{create} & Create a new resource \\
      \texttt{update} & Modify an existing resource (full replace) \\
      \texttt{patch} & Partially modify a resource \\
      \texttt{delete} & Delete a resource \\
      \texttt{deletecollection} & Delete multiple resources at once \\
      \texttt{*} & All verbs (wildcard) \\

    \cmdhead{Check which group a resource belongs to}
    \begin{cheatcodebash}
    k api-resources    # GROUP column shows the apiGroup
    \end{cheatcodebash}


    \cmdhead{Variations}
    \begin{cheatcodebash}
    # ClusterRoleBinding for a group: read-only across cluster
    k create clusterrole view-all --verb=get,list,watch --resource='*'
    k create clusterrolebinding view-all-binding \
      --clusterrole=view-all --group=dev-team
    \end{cheatcodebash}
    \begin{cheatcodebash}
    # ClusterRoleBinding for a single user: manage pods
    k create clusterrole pod-mgr \
      --verb=get,list,watch,create,update,delete --resource=pods
    k create clusterrolebinding pod-mgr-binding \
      --clusterrole=pod-mgr --user=jane
    \end{cheatcodebash}
    \begin{cheatcodebash}
    # Role in a specific namespace
    k create role deploy-mgr \
      --verb=get,list,create,update --resource=deployments -n staging
    k create rolebinding deploy-mgr-binding \
      --role=deploy-mgr --user=jane -n staging
    \end{cheatcodebash}
