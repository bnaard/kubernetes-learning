        \tcbsubtitle{Fix a Broken Node}
        {\fontsize{6.5}{7.5}\selectfont\sffamily
        \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
          \item \texttt{k get nodes} — identify \texttt{NotReady} node
          \item \texttt{ssh <node>}
          \item \texttt{systemctl status kubelet} — check if running
          \item \texttt{journalctl -u kubelet | tail -50} — find error
          \item Fix config: wrong \texttt{--config} path, bad cert path, wrong \texttt{staticPodPath}
          \item \texttt{systemctl restart kubelet}
          \item \texttt{systemctl enable kubelet} — ensure it survives reboot
          \item Verify: \texttt{k get nodes} — node becomes \texttt{Ready}
        \end{enumerate}}

      \tcbitem
          \tcbsubtitle{Fix a Broken Control Plane}
          {\fontsize{6.5}{7.5}\selectfont\sffamily
          \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
            \item \texttt{k get pods -n kube-system} — or \texttt{crictl ps} if API is down
            \item Check static pod manifests in \texttt{/etc/kubernetes/manifests/}
            \item \texttt{crictl logs <id>} — find the error
            \item Common fixes: wrong image tag, bad volume mount path, typo in \texttt{--etcd-servers}, wrong cert path
            \item Save manifest — kubelet auto-restarts the pod
            \item Verify: \texttt{k get pods -n kube-system}
          \end{enumerate}}

      \tcbitem
          \tcbsubtitle{Upgrade a Cluster (kubeadm)}
          {\fontsize{6.5}{7.5}\selectfont\sffamily
          Control plane:
          \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
            \item \texttt{apt-mark unhold kubeadm} — allow apt to upgrade it
            \item \texttt{apt update \&\& apt install -y kubeadm=1.X.Y-1.1}
            \item \texttt{kubeadm upgrade plan}
            \item \texttt{kubeadm upgrade apply v1.X.Y}
            \item \texttt{apt install -y kubelet=1.X.Y-1.1 kubectl=1.X.Y-1.1}
            \item \texttt{systemctl daemon-reload \&\& systemctl restart kubelet}
          \end{enumerate}
          Worker nodes:
          \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
            \item \texttt{k drain <node> --ignore-daemonsets --delete-emptydir-data}
            \item SSH to node, then: \texttt{kubeadm upgrade node}
            \item \texttt{apt install -y kubelet=1.X.Y-1.1 kubectl=1.X.Y-1.1}
            \item \texttt{systemctl daemon-reload \&\& systemctl restart kubelet}
            \item \texttt{k uncordon <node>}
          \end{enumerate}}

          \tcbsubtitle{Troubleshoot a Service}
          {\fontsize{6.5}{7.5}\selectfont\sffamily
          \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
            \item \texttt{k get endpoints <svc>} — empty = selector mismatch
            \item Compare \texttt{svc} selector vs pod labels
            \item Check \texttt{targetPort} matches container port
            \item \texttt{k run tmp --rm -it --image=busybox -- wget -qO- <svc>:<port>}
            \item Check NetworkPolicy: \texttt{k get netpol -A}
          \end{enumerate}}

          \tcbsubtitle{Create ServiceAccount + Bind Role}
          {\fontsize{6.5}{7.5}\selectfont\sffamily
          \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
            \item \texttt{k create sa deploy-admin -n <ns>}
            \item \texttt{k create clusterrole cr --verb=get --resource=pods}
            \item \texttt{k create clusterrolebinding crb --clusterrole=cr --serviceaccount=<ns>:deploy-admin}
            \item Verify: \texttt{k auth can-i get pods --as system:serviceaccount:<ns>:deploy-admin}
