% === PersistentVolume =====================================================
\tcbsubtitle{PersistentVolume (PV)}
\begin{cheatcodebashbold}
k get pv    # check RECLAIM POLICY, STATUS
k create pv <pv-name> --capacity=<size> --access-modes=<mode> --host-path=<path>
\end{cheatcodebashbold}

\begin{cheatcodebash}
apiVersion: v1
kind: PersistentVolume
metadata: {name: my-pv}
spec:
  capacity: {storage: 10Gi}
  accessModes:
  - ReadWriteOnce   # RWO: single node r/w
  # ReadWriteMany   # RWX: many nodes r/w
  # ReadOnlyMany    # ROX: many nodes r/o
  persistentVolumeReclaimPolicy: Retain
  # Retain   = keep PV after PVC delete
  # Delete   = delete underlying storage
  # Recycle  = deprecated
  storageClassName: manual
  hostPath: {path: /mnt/data, type: Directory} 
# nfs: {server: 1.2.3.4, path: /x}
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain persistentvolume.spec --recursive
# docs: "persistent volumes"
\end{cheatcodebash}

% === PersistentVolumeClaim ================================================
\tcbsubtitle{PersistentVolumeClaim (PVC)}

\begin{cheatcodebashbold}
k get pvc    # STATUS: Bound|Pending|Lost
k create pv my-local-pv --capacity=5Gi --access-modes=ReadWriteOnce --host-path=/data/config
\end{cheatcodebashbold}

\begin{cheatcodebash}
apiVersion: v1
kind: PersistentVolumeClaim
metadata: {name: my-pvc, namespace: default}
spec:
  accessModes: [ReadWriteOnce]
  storageClassName: manual  # "" = no dynamic prov.
  resources:
    requests: {storage: 5Gi}
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain persistentvolumeclaim.spec --recursive
# docs: "persistent volumes"
\end{cheatcodebash}

% === StorageClass (Dynamic Provisioning) ==================================
\tcbsubtitle{StorageClass (Dynamic Provisioning)}
\begin{cheatcodebash}
k get sc               # list StorageClasses
\end{cheatcodebash}
\begin{cheatcodebash}
apiVersion: storage.k8s.io/v1
kind: StorageClass
metadata:
  name: fast
  annotations:
    storageclass.kubernetes.io/is-default-class: "true"
provisioner: rancher.io/local-path
reclaimPolicy: Delete
allowVolumeExpansion: true
volumeBindingMode: WaitForFirstConsumer
\end{cheatcodebash}
\begin{cheatcodebash}
spec:  # PVC with dynamic provisioning
  storageClassName: fast  # matches SC name
  resources:
    requests: {storage: 5Gi}
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain storageclass --recursive
# docs: "storage classes"
\end{cheatcodebash}
