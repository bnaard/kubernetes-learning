        % === StorageClass (Dynamic Provisioning) ==================================
        \tcbsubtitle{StorageClass (Dynamic Provisioning)}
        \begin{cheatcodebash}
        k get sc               # list StorageClasses
        \end{cheatcodebash}
        \begin{cheatcodebash}
        apiVersion: storage.k8s.io/v1
        kind: StorageClass
        metadata:
          name: fast
          annotations:
            storageclass.kubernetes.io/is-default-class: "true"
        provisioner: rancher.io/local-path
        reclaimPolicy: Delete
        allowVolumeExpansion: true
        volumeBindingMode: WaitForFirstConsumer
        \end{cheatcodebash}
        \begin{cheatcodebash}
        spec:  # PVC with dynamic provisioning
          storageClassName: fast  # matches SC name
          resources:
            requests: {storage: 5Gi}
        \end{cheatcodebash}

        \cmdhead{Help}
        \begin{cheatcodebash}
        k explain storageclass --recursive
        \end{cheatcodebash}
        \cmdhead{\url{https://kubernetes.io/docs/}}
        \begin{cheatcodebash}
        search: "storage classes"
        \end{cheatcodebash}

        % === PersistentVolume =====================================================
        \tcbsubtitle{PersistentVolume (PV)}
        \begin{cheatcodebashbold}
        k get pv    # check RECLAIM POLICY, STATUS
        k create pv <pv-name> --capacity=<size> --access-modes=ReadWriteOnce --host-path=<path>
        \end{cheatcodebashbold}

        \begin{cheatcodebash}
        apiVersion: v1
        kind: PersistentVolume
        metadata: {name: my-pv}
        spec:
          capacity: {storage: 10Gi}
          accessModes:
          - ReadWriteOnce   # RWO: single node r/w
          # ReadWriteMany   # RWX: many nodes r/w
          # ReadOnlyMany    # ROX: many nodes r/o
          persistentVolumeReclaimPolicy: Retain
          # Retain   = keep PV after PVC delete
          # Delete   = delete underlying storage
          # Recycle  = deprecated
          storageClassName: manual  # static provisioning (pre-created PV)
          hostPath: {path: /mnt/data, type: Directory}
        # nfs: {server: 1.2.3.4, path: /x}
        \end{cheatcodebash}

        \cmdhead{Help}
        \begin{cheatcodebash}
        k explain persistentvolume.spec --recursive
        \end{cheatcodebash}
        \cmdhead{\url{https://kubernetes.io/docs/}}
        \begin{cheatcodebash}
        search: "persistent volumes"
        \end{cheatcodebash}

        % === PersistentVolumeClaim ================================================
        \tcbsubtitle{PersistentVolumeClaim (PVC)}

        \begin{cheatcodebashbold}
        k get pvc    # STATUS: Bound|Pending|Lost
        # No "k create pvc" — use YAML (apply -f)
        \end{cheatcodebashbold}

        \begin{cheatcodebash}
        apiVersion: v1
        kind: PersistentVolumeClaim
        metadata: {name: my-pvc, namespace: default}
        spec:
          accessModes: [ReadWriteOnce]
          storageClassName: manual  # non-existent SC → static prov. ("" works too)
          resources:
            requests: {storage: 5Gi}
        \end{cheatcodebash}

        \cmdhead{Help}

        % === Pod Volumes & Volume Mounts ================================================
        \tcbsubtitle{Pod Volumes \& Volume Mounts}
        \begin{cheatcodebash}
        spec:                  # <pod>.spec
          volumes:
          - name: pvc-vol
            persistentVolumeClaim:
              claimName: my-pvc
          - name: tmp
            emptyDir: {}           # ephemeral, lost on restart
          - name: cfg
            configMap: {name: my-cm}
          - name: sec
            secret: {secretName: my-secret}
          - name: host
            hostPath: {path: /data, type: Directory}
          containers:
          - volumeMounts:
            - {name: pvc-vol, mountPath: /var/data}
            - {name: cfg, mountPath: /etc/config, readOnly: true}
        \end{cheatcodebash}

        \cmdhead{Help}
        \begin{cheatcodebash}
        k explain pod.spec.volumes --recursive
        k explain pod.spec.containers.volumeMounts --recursive
        \end{cheatcodebash}
        \cmdhead{\url{https://kubernetes.io/docs/}}
        \begin{cheatcodebash}
        search: "volumes"
        \end{cheatcodebash}
