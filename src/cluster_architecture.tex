% === High Availability ====================================================
\tcbsubtitle{High Availability}
\begin{itemize}[nosep, leftmargin=1em, label={--}]
  \item \textbf{Stacked etcd}: etcd co-located on ctrl-plane nodes (default)
  \item \textbf{External etcd}: separate etcd cluster (more resilient)
  \item Requires \textbf{loadbalancer (LB)} in front of multiple API servers
\end{itemize}
\includegraphics[width=\linewidth]{ha_cluster_structure.png}
\vspace{1ex}

Initialize the first control plane node (↓ outputs join commands)
\begin{cheatcodebash}
kubeadm init --control-plane-endpoint="<LB_IP>:6443" --upload-certs
\end{cheatcodebash}

\vspace{1ex}

{Join additional {\color{red}control-plane} and {\color{teal}worker} nodes:}
\begin{cheatcodetex}
\ttfamily\fontsize{6.5}{7.5}\selectfont
\textcolor{red}{kubeadm join <LB>:6443 -{}-control-plane -{}-certificate-key <key>}\\
\textcolor{teal}{kubeadm join <LB\_IP>:6443 ...}
\end{cheatcodetex}

% === Key Paths ============================================================
\tcbsubtitle{Key Paths}
\begin{cheatcodebashbold}
/etc/kubernetes/manifests/   # static pod manifests
/var/lib/kubelet/config.yaml # kubelet config
/etc/kubernetes/pki/         # cluster TLS certs
/etc/kubernetes/admin.conf   # admin kubeconfig
/etc/cni/net.d/              # CNI plugin config
\end{cheatcodebashbold}
