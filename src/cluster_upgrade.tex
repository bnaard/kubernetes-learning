% === Cluster Upgrade ======================================================
\tcbsubtitle{Cluster Upgrade (kubeadm)}
\begin{cheatcodebash}
# --- Control Plane ---
apt-mark unhold kubeadm && apt install -y kubeadm=1.X.Y-1.1
kubeadm upgrade plan
kubeadm upgrade apply v1.X.Y
apt install -y kubelet=1.X.Y-1.1 kubectl=1.X.Y-1.1
systemctl daemon-reload && systemctl restart kubelet

# --- Each Worker Node ---
kubectl drain <node> --ignore-daemonsets --delete-emptydir-data
# SSH to node, then:
kubeadm upgrade node
apt install -y kubelet=1.X.Y-1.1 kubectl=1.X.Y-1.1
systemctl daemon-reload && systemctl restart kubelet
kubectl uncordon <node>
\end{cheatcodebash}
