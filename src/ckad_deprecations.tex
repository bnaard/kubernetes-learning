  \tcbsubtitle{Migration Workflow}
  {\fontsize{6.5}{7.5}\selectfont\sffamily
  \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
    \item Identify deprecated APIs: check metrics or use \texttt{kubectl convert}
    \item Update manifests to new API version
    \item \texttt{k apply -f updated-manifest.yaml}
    \item Verify: \texttt{k get <resource>} --- confirm running with new API
  \end{enumerate}}
