% === CRDs & Operators =====================================================
\tcbsubtitle{CRDs \& Operators}
\begin{cheatcodebash}
k get crd                       # list all CRDs
k describe crd <name>           # schema + versions
k get <custom-resource-kind>    # list custom objects
\end{cheatcodebash}
\begin{cheatcodebash}
# Operators extend K8s with custom controllers
# Install via Helm or OLM (Operator Lifecycle Manager):
k get operators                 # if OLM installed
\end{cheatcodebash}

\begin{cheatcodebash}
### Helpers:
k explain customresourcedefinition.spec --recursive
# docs: "custom resources"
\end{cheatcodebash}
