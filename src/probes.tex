% === Metrics Server =======================================================

  \tcbitem
    \tcbsubtitle{Metrics Server}

    \begin{cheatcodebash}
k top nodes                       # CPU/mem per node
k top pods                        # CPU/mem per pod
k top pods --containers           # per-container breakdown
k top pods --sort-by=cpu          # find heaviest consumer
k top pods --sort-by=memory -A    # across all namespaces
    \end{cheatcodebash}

    \cmdhead{Helpers}
    \begin{cheatcodebash}
    k explain node.status.allocatable --recursive
    \end{cheatcodebash}
    \cmdhead{\url{https://kubernetes.io/docs/}}
    \begin{cheatcodebash}
    search: "metrics server"
    \end{cheatcodebash}

  \tcbitem
    \tcbsubtitle{Probes}

        % === Probes ===============================================================
        {\small\sffamily
        \begin{tblr}{
          colspec={lX},
          hline{2}={0.4pt},
          rows={rowsep=0.5pt},
          row{1}={font=\bfseries},
        }
        Probe & Triggers \\
        liveness & restart container \\
        readiness & remove from Service endpoints \\
        startup & disables others during slow start \\
        \end{tblr}}

      \begin{cheatcodebash}
        spec:               # <pod>.spec
          containers:
          - livenessProbe:  # OR readinessProbe/startupProbe
              httpGet:
                path: /healthz
                port: 8080
              # OR: tcpSocket: {port: 8080}
              # OR: exec: {command: [cat, /tmp/healthy]}
              initialDelaySeconds: 10  # wait before first check
              periodSeconds: 5         # interval between checks
              failureThreshold: 3      # failures before action
              successThreshold: 1      # successes to recover
              timeoutSeconds: 1        # max wait per check
        \end{cheatcodebash}

    \cmdhead{Helpers}
    \begin{cheatcodebash}
    k explain pod.spec.containers.livenessProbe --recursive
    k explain pod.spec.containers.readinessProbe --recursive
    \end{cheatcodebash}
    \cmdhead{\url{https://kubernetes.io/docs/}}
    \begin{cheatcodebash}
    search: "configure liveness readiness startup probes"
    \end{cheatcodebash}
