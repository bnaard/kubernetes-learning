
% === Namespaces ===========================================================
\cmdhead{Create and list}
\begin{cheatcodebashbold}
k create ns <ns>
k get all -n <ns>   # all resources in <ns>
k get all -A        # all resources in all namespaces
\end{cheatcodebashbold}

\cmdhead{Delete namespace + all resources}
\begin{cheatcodebashbold}
k delete ns <ns>
\end{cheatcodebashbold}

\cmdhead{Cross-namespace DNS}
\begin{cheatcodebashbold}
# <svc>.<ns>.svc.cluster.local
\end{cheatcodebashbold}

\cmdhead{List resources}
\begin{cheatcodebash}
k api-resources --namespaced=false # cluster-scoped
k api-resources --namespaced=true  # namespace-scoped
\end{cheatcodebash}
