% === ConfigMaps ===========================================================
\tcbsubtitle{ConfigMaps}
\begin{cheatcode}
k create cm <name> --from-literal=key=val
k create cm <name> --from-file=config.conf
k create cm <name> --from-env-file=.env
k get cm <name> -o yaml
k describe cm <name>
\end{cheatcode}

ConfigMap YAML:
\begin{cheatcode}
apiVersion: v1
kind: ConfigMap
metadata: {name: my-cm}
data:
  key1: value1
  config.conf: |
    server=localhost
    port=8080
\end{cheatcode}

% === Secrets ==============================================================
\tcbsubtitle{Secrets}
\begin{cheatcode}
# Generic (Opaque) - base64 encoded:
k create secret generic <name> \
  --from-literal=pwd=s3cr3t
# TLS:
k create secret tls <name> \
  --cert=tls.crt --key=tls.key
# Docker registry:
k create secret docker-registry <name> \
  --docker-server=reg.io \
  --docker-username=u --docker-password=p
# Decode value:
k get secret <name> \
  -o jsonpath='{.data.pwd}' | base64 -d
\end{cheatcode}

Secret YAML (values are base64):
\begin{cheatcode}
apiVersion: v1
kind: Secret
metadata: {name: my-sec}
type: Opaque
data:
  pwd: czNjcjN0   # echo -n 's3cr3t' | base64
stringData:       # plaintext, auto-encoded
  apikey: my-key
\end{cheatcode}

% === Using ConfigMaps & Secrets in Pods ===================================
\tcbsubtitle{Use in Pods}
\begin{cheatcode}
# Individual key as env var:
env:
- name: ENV_VAR
  valueFrom:
    configMapKeyRef: {name: my-cm, key: key1}
- name: SECRET_VAR
  valueFrom:
    secretKeyRef: {name: my-sec, key: pwd}
# All keys as env vars:
envFrom:
- configMapRef: {name: my-cm}
- secretRef:    {name: my-sec}
# Mount as volume (each key = a file):
volumes:
- name: cfg-vol
  configMap: {name: my-cm}
- name: sec-vol
  secret: {secretName: my-sec}
containers:
- volumeMounts:
  - {name: cfg-vol, mountPath: /etc/config}
  - {name: sec-vol, mountPath: /etc/secrets,
     readOnly: true}
\end{cheatcode}
