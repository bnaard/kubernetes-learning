    % === ConfigMaps ===========================================================
    \tcbsubtitle{ConfigMaps}
    \cmdhead{Imperative}
    \begin{cheatcodebashbold}
    k create cm <name> --from-literal=key=val
    k create cm <name> --from-file=config.conf # pairs in file
    k create cm <name> --from-env-file=.env
    k get cm <name> -o yaml
    k describe cm <name>
    \end{cheatcodebashbold}


    \cmdhead{Spec Essentials}
    \begin{cheatcodebash}
    apiVersion: v1
    kind: ConfigMap
    metadata: {name: my-cm}
    data:
      key1: value1
      config.conf: |
        server=localhost
        port=8080
    \end{cheatcodebash}

    \cmdhead{Help}
    \begin{cheatcodebash}
    k explain configmap
    \end{cheatcodebash}

    \cmdhead{\url{https://kubernetes.io/docs/}}
    \begin{cheatcodebash}
    search: "configmaps"
    \end{cheatcodebash}


  \tcbitem

    % === Secrets ==============================================================
    \tcbsubtitle{Secrets}
    {\footnotesize \warning{etcd stores secrets base64 \textbf{encoded}}}

    \cmdhead{Create Generic (Opaque) - base64 encoded}
    \begin{cheatcodebashbold}
    k create secret generic <name> --from-literal=pwd=s3cr3t
    \end{cheatcodebashbold}

    \cmdhead{Create TLS (certificate + private key, eg for Ingress)}
    \begin{cheatcodebashbold}
    k create secret tls <name> --cert=tls.crt --key=tls.key
    \end{cheatcodebashbold}

    \cmdhead{Docker registry}
    \begin{cheatcodebash}
    k create secret docker-registry <name> \
      --docker-server=reg.io \
      --docker-username=u --docker-password=p
    \end{cheatcodebash}

    \cmdhead{Decode value}
    \begin{cheatcodebashbold}
    k get secret <name> -o jsonpath='{.data.pwd}' | base64 -d
    \end{cheatcodebashbold}

    \cmdhead{Spec Essentials}
    \begin{cheatcodebash}
    apiVersion: v1
    kind: Secret
    metadata: {name: my-sec}
    type: Opaque
    data:
      pwd: czNjcjN0   # echo -n 's3cr3t' | base64
    stringData:       # plaintext, auto-encoded
      apikey: my-key
    \end{cheatcodebash}

    \cmdhead{Help}
    \begin{cheatcodebash}
    k explain secret
    \end{cheatcodebash}

    \cmdhead{\url{https://kubernetes.io/docs/}}
    \begin{cheatcodebash}
    search: "secrets"
    \end{cheatcodebash}


  \tcbitem

    % === Using ConfigMaps & Secrets in Pods ===================================
    \tcbsubtitle{Usage in Pods}
    \cmdhead{Individual key as env var}
    \begin{cheatcodebashbold}
    env:                   # <pod>.spec.containers[].env[]
    - name: ENV_VAR
      valueFrom:
        configMapKeyRef: {name: my-cm, key: key1}
    - name: SECRET_VAR
      valueFrom:
        secretKeyRef: {name: my-sec, key: pwd}
    \end{cheatcodebashbold}

    \cmdhead{All keys as env vars}
    \begin{cheatcodebash}
    envFrom:               # <pod>.spec.containers[].envFrom[]
    - configMapRef: {name: my-cm}
    - secretRef:    {name: my-sec}
    \end{cheatcodebash}

    \cmdhead{Mount as volume (each key = a file)}
    \begin{cheatcodebashbold}
    volumes:               # <pod>.spec.volumes[]
    - name: cfg-vol
      configMap: {name: my-cm}
    - name: sec-vol
      secret: {secretName: my-sec}
    containers:
    - volumeMounts:
      - {name: cfg-vol, mountPath: /etc/config}
      - {name: sec-vol, mountPath: /etc/secrets,
        readOnly: true}
    \end{cheatcodebashbold}

    \cmdhead{Help}
    \begin{cheatcodebash}
    k explain pod.spec.containers.envFrom --recursive
    k explain pod.spec.containers.env.valueFrom --recursive
    k explain pod.spec.volumes --recursive
    \end{cheatcodebash}

    \cmdhead{\url{https://kubernetes.io/docs/}}
    \begin{cheatcodebash}
    search: "configure pod configmap"
    \end{cheatcodebash}
