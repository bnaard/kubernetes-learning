% === ConfigMaps ===========================================================
\tcbsubtitle{ConfigMaps}
\begin{cheatcodebashbold}
k create cm <name> --from-literal=key=val
k create cm <name> --from-file=config.conf
k create cm <name> --from-env-file=.env
k get cm <name> -o yaml
k describe cm <name>
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
### Create configmap / secret / service account
k create cm <name> --from-literal=k=v
k create cm <name> --from-file=<filepath> # pairs in file
k create cm <name> --from-env-file=<filepath>
k create secret generic <name> --from-literal=pwd=s3cr3t
k create sa <name> -n <ns>
\end{cheatcodebashbold}


\begin{cheatcodebash}
apiVersion: v1
kind: ConfigMap
metadata: {name: my-cm}
data:
  key1: value1
  config.conf: |
    server=localhost
    port=8080
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain configmap
# docs: "configmaps"
\end{cheatcodebash}

% === Secrets ==============================================================
\tcbsubtitle{Secrets}
{\footnotesize \warning{etcd stores secrets base64 \textbf{encoded}}}

\begin{cheatcodebashbold}
# Generic (Opaque) - base64 encoded:
k create secret generic <name> --from-literal=pwd=s3cr3t
# TLS (certificate + private key, eg for Ingress)
k create secret tls <name> --cert=tls.crt --key=tls.key
\end{cheatcodebashbold}

\begin{cheatcodebash}
# Docker registry:
k create secret docker-registry <name> \
  --docker-server=reg.io \
  --docker-username=u --docker-password=p
\end{cheatcodebash}

\begin{cheatcodebashbold}
# Decode value:
k get secret <name> -o jsonpath='{.data.pwd}' | base64 -d
\end{cheatcodebashbold}

\begin{cheatcodebash}
apiVersion: v1
kind: Secret
metadata: {name: my-sec}
type: Opaque
data:
  pwd: czNjcjN0   # echo -n 's3cr3t' | base64
stringData:       # plaintext, auto-encoded
  apikey: my-key
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain secret
# docs: "secrets"
\end{cheatcodebash}

% === Using ConfigMaps & Secrets in Pods ===================================
\tcbsubtitle{Use in Pods}
\begin{cheatcodebashbold}
# Individual key as env var:
env:                   # <pod>.spec.containers[].env[]
- name: ENV_VAR
  valueFrom:
    configMapKeyRef: {name: my-cm, key: key1}
- name: SECRET_VAR
  valueFrom:
    secretKeyRef: {name: my-sec, key: pwd}
\end{cheatcodebashbold}

\begin{cheatcodebash}
# All keys as env vars:
envFrom:               # <pod>.spec.containers[].envFrom[]
- configMapRef: {name: my-cm}
- secretRef:    {name: my-sec}
\end{cheatcodebash}

\begin{cheatcodebashbold}
# Mount as volume (each key = a file):
volumes:               # <pod>.spec.volumes[]
- name: cfg-vol
  configMap: {name: my-cm}
- name: sec-vol
  secret: {secretName: my-sec}
containers:
- volumeMounts:
  - {name: cfg-vol, mountPath: /etc/config}
  - {name: sec-vol, mountPath: /etc/secrets,
     readOnly: true}
\end{cheatcodebashbold}

\begin{cheatcodebash}
### Exam helpers:
k explain pod.spec.containers.envFrom --recursive
k explain pod.spec.containers.env.valueFrom --recursive
k explain pod.spec.volumes --recursive
# docs: "configure pod configmap"
\end{cheatcodebash}
