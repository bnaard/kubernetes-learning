% === Helm =================================================================
\tcbsubtitle{Helm}
\begin{cheatcodebash}
helm repo add <name> <url>       # register chart repo
helm repo update                 # refresh index
helm search repo <keyword>       # find charts
helm install <release> <chart> \
  -n <ns> --create-namespace \
  -f values.yaml --set key=val   # deploy
helm upgrade <release> <chart> -f vals.yaml
helm rollback <release> [revision]
helm uninstall <release> -n <ns> # delete release
helm list [-A]                   # show releases
helm status <release>            # health + notes
helm get values <release>        # current overrides
helm template <release> <chart>  # dry-run
helm show values <chart>         # inspect defaults
\end{cheatcodebash}

% === Kustomize ============================================================
\tcbsubtitle{Kustomize}
\begin{cheatcodebash}
# kustomization.yaml:
apiVersion: kustomize.config.k8s.io/v1beta1
kind: Kustomization
resources:
- deployment.yaml
- service.yaml
namespace: my-ns
images:
- name: nginx
  newTag: 1.25
patches:
- patch: |-
    - op: replace
      path: /spec/replicas
      value: 3
  target: {kind: Deployment, name: web}
\end{cheatcodebash}
\begin{cheatcodebash}
# Apply:
kubectl apply -k <dir>
kubectl kustomize <dir>        # preview only
\end{cheatcodebash}

% === CRDs & Operators =====================================================
\tcbsubtitle{CRDs \& Operators}
\begin{cheatcodebash}
k get crd                       # list all CRDs
k describe crd <name>           # schema + versions
k get <custom-resource-kind>    # list custom objects
\end{cheatcodebash}
\begin{cheatcodebash}
# Operators extend K8s with custom controllers
# Install via Helm or OLM:
k get operators                 # if OLM installed
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain customresourcedefinition.spec --recursive
# docs: "custom resources"
\end{cheatcodebash}
