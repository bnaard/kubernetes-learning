% === Helm =================================================================
\tcbsubtitle{Helm}
\begin{cheatcode}
helm repo add <name> <url>
helm repo update
helm search repo <keyword>
helm install <release> <chart> \
  -n <ns> --create-namespace \
  -f values.yaml --set key=val
helm upgrade <release> <chart> -f vals.yaml
helm rollback <release> [revision]
helm uninstall <release> -n <ns>
helm list [-A]
helm status <release>
helm get values <release>
helm template <release> <chart>  # dry-run
helm show values <chart>         # inspect defaults
\end{cheatcode}

% === Kustomize ============================================================
\tcbsubtitle{Kustomize}
\begin{cheatcode}
# kustomization.yaml:
apiVersion: kustomize.config.k8s.io/v1beta1
kind: Kustomization
resources:
- deployment.yaml
- service.yaml
namespace: my-ns
images:
- name: nginx
  newTag: 1.25
patches:
- patch: |-
    - op: replace
      path: /spec/replicas
      value: 3
  target: {kind: Deployment, name: web}
# Apply:
kubectl apply -k <dir>
kubectl kustomize <dir>        # preview only
\end{cheatcode}
