% === Jobs =================================================================
\tcbsubtitle{Jobs}
\cmdhead{Imperative}
\begin{cheatcodebash}
k create job <name> --image=busybox -- echo hi $do > j.yaml
k get jobs              # List jobs
k logs job/<name>       # Get log of job <name>
\end{cheatcodebash}

\cmdhead{Spec Essentials}
\begin{cheatcodebash}
spec:                   # <job>.spec
  completions: 3        # total successful completions
  parallelism: 2        # concurrent pods
  backoffLimit: 4       # retries before fail
  ttlSecondsAfterFinished: 60 # time to live job obj.
  template:
    spec:
      restartPolicy: Never  # or OnFailure
      containers:
      - name: job
        image: busybox
        command: ["echo", "done"]
\end{cheatcodebash}

\cmdhead{Help}
\begin{cheatcodebash}
k explain job.spec --recursive
\end{cheatcodebash}

\cmdhead{\url{https://kubernetes.io/docs/}}
\begin{cheatcodebash}
search: "jobs"
\end{cheatcodebash}

% === CronJobs =============================================================
\tcbsubtitle{CronJobs}

\cmdhead{Imperative}
\begin{cheatcodebash}
k create cj <name> --image=busybox --schedule="*/5 * * * *" -- echo hi
k get cj
k create job manual-run --from=cronjob/<name>
\end{cheatcodebash}

\cmdhead{Spec Essentials}
\begin{cheatcodebash}
spec:                       # <cronjob>.spec
  schedule: "0 2 * * *"
  concurrencyPolicy: Allow  # or Forbid or Replace
  successfulJobsHistoryLimit: 3
  failedJobsHistoryLimit: 1
  startingDeadlineSeconds: 60
  jobTemplate:
    spec: ...               # same as Job spec
\end{cheatcodebash}

\cmdhead{Help}
\begin{cheatcodebash}
k explain cronjob.spec --recursive
\end{cheatcodebash}

\cmdhead{\url{https://kubernetes.io/docs/}}
\begin{cheatcodebash}
search: "cron jobs"
\end{cheatcodebash}
