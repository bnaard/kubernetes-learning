% === HPA ==================================================================
\tcbsubtitle{HPA (Horizontal Pod Autoscaler)}
\footnotesize{\note{Requires Metrics Server}}

\begin{cheatcodebashbold}
#  creates a HPA for deployment <name>:
#  - Watches the average CPU utilization across all pods                                                                                                                                                 
#  - If avg CPU > 50% of requested CPU, adds pods, max 10                                                                                                                   
#  - If avg CPU < 50%, scales down (removes pods, min 2)    
k autoscale deploy <name> --cpu-percent=50 --min=2 --max=10
\end{cheatcodebashbold}
\begin{cheatcodebashbold}
k get hpa
k describe hpa <name>
\end{cheatcodebashbold}
\begin{cheatcodebash}
# HPA YAML (autoscaling/v2):
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
spec:
  scaleTargetRef:              # what to scale
    apiVersion: apps/v1
    kind: Deployment
    name: my-deploy
  minReplicas: 2               # never fewer than 2
  maxReplicas: 10              # never more than 10
  metrics:
  - type: Resource             # built-in CPU/memory
    resource:
      name: cpu
      target:
        type: Utilization      # % of requests
        averageUtilization: 50
  - type: Resource
    resource:
      name: memory
      target:
        type: AverageValue     # absolute value
        averageValue: 200Mi
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain horizontalpodautoscaler.spec --recursive
# docs: "horizontal pod autoscaling"
\end{cheatcodebash}

% === VPA ==================================================================
\tcbsubtitle{VPA (Vertical Pod Autoscaler)}
\begin{cheatcodebash}
k get vpa
k describe vpa <name>  # check Recommendation
\end{cheatcodebash}

\begin{cheatcodebash}
# VPA adjusts resource requests/limits
# Requires VPA controller installed
apiVersion: autoscaling.k8s.io/v1
kind: VerticalPodAutoscaler
metadata: {name: my-vpa}
spec:
  targetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: my-deploy
  updatePolicy:
    updateMode: "Auto"
    # Off      = recommend only (k describe vpa)
    # Initial  = set on new pods only
    # Recreate = evict + restart pods
    # Auto     = same as Recreate currently
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain verticalpodautoscaler.spec --recursive
# docs: "vertical pod autoscaler"
\end{cheatcodebash}
