% === HPA ==================================================================
\tcbsubtitle{HPA (Horizontal Pod Autoscaler)}
\begin{cheatcode}
# Requires Metrics Server!
k autoscale deploy <name> \
  --cpu-percent=50 --min=2 --max=10
k get hpa
k describe hpa <name>
# HPA YAML (autoscaling/v2):
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: my-deploy
  minReplicas: 2
  maxReplicas: 10
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 50
  - type: Resource
    resource:
      name: memory
      target:
        type: AverageValue
        averageValue: 200Mi
\end{cheatcode}

% === VPA ==================================================================
\tcbsubtitle{VPA (Vertical Pod Autoscaler)}
\begin{cheatcode}
# VPA adjusts resource requests/limits
# Requires VPA controller installed
apiVersion: autoscaling.k8s.io/v1
kind: VerticalPodAutoscaler
metadata: {name: my-vpa}
spec:
  targetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: my-deploy
  updatePolicy:
    updateMode: "Auto"
    # Off      = recommend only (k describe vpa)
    # Initial  = set on new pods only
    # Recreate = evict + restart pods
    # Auto     = same as Recreate currently
k get vpa
k describe vpa <name>  # check Recommendation
\end{cheatcode}
