% === Pod Debugging ========================================================
\tcbsubtitle{Pod Debugging}
\begin{cheatcodebash}
k get pods -A -o wide
k describe pod <name> -n <ns>   # check Events!
k logs <pod> [-c <container>]
k logs <pod> --previous         # or -p crashed container
k exec -it <pod> -- sh
k exec -it <pod> -c <c> -- bash
\end{cheatcodebash}
\begin{cheatcodebash}
# Ephemeral debug container (K8s 1.23+):
k debug -it <pod> --image=busybox --target=<container>
# Copy pod for debugging / test modified spec 
k debug <pod> -it --copy-to=<pod>-debug --image=busybox
\end{cheatcodebash}


\end{minipage}\hfill

% === Common Pod States ====================================================
\tcbsubtitle{Common Pod States}
{\small\sffamily
\begin{tblr}{
  colspec={lX},
  hline{2}={0.4pt},
  rows={rowsep=0.5pt},
  row{1}={font=\bfseries},
}
Status & Likely Cause \& Fix \\
Pending & No node fits: check taints, resources, PVC \\
CrashLoopBackOff & App crashes: \texttt{k logs -{}-previous} \\
OOMKilled & Mem limit exceeded: raise \texttt{limits.memory} \\
ImagePullBackOff & Bad image name or no pull secret \\
Error & Non-zero exit: \texttt{k logs} \\
Terminating & Stuck finalizer: \texttt{k delete -{}-force} \\
ContainerCreating & Volume or image pull in progress \\
\end{tblr}}

% === Node Debugging ========================================================
\tcbsubtitle{Node Debugging}
\begin{cheatcodebash}
k get nodes -o wide
k describe node <name>      # Conditions, Events
k cordon <node>             # mark unschedulable
k drain <node> --ignore-daemonsets --delete-emptydir-data
k uncordon <node>
\end{cheatcodebash}
\begin{cheatcodebash}
# On the node (SSH):
systemctl status kubelet
systemctl restart kubelet
journalctl -u kubelet -f
journalctl -u kubelet | grep -i error
systemctl status containerd
systemctl cat kubelet       # see config path
\end{cheatcodebash}

\end{minipage}\hfill

% === Control Plane Debug ==================================================
\tcbsubtitle{Control Plane Debug}
\begin{cheatcodebash}
# Static pods in kube-system:
k get pods -n kube-system
k logs kube-apiserver-<node> -n kube-system
k logs etcd-<node> -n kube-system
\end{cheatcodebash}

\begin{cheatcodebash}
# Edit control plane component:
vi /etc/kubernetes/manifests/kube-apiserver.yaml
\end{cheatcodebash}
\begin{cheatcodebash}
# Kubelet config path:
systemctl cat kubelet | grep '\-\-config'
# Usually: /var/lib/kubelet/config.yaml
\end{cheatcodebash}
\begin{cheatcodebash}
# If API server unreachable:
crictl ps                   # list containers on node
crictl logs <container-id>
\end{cheatcodebash}
\begin{cheatcodebash}
# Kubeconfig:
export KUBECONFIG=/etc/kubernetes/admin.conf
\end{cheatcodebash}

% === Service Debugging =====================================================
\tcbsubtitle{Service \& Networking Debugging}
\begin{cheatcodebash}
# Check if svc routes to pods:
k get endpoints <svc>       # empty = label mismatch
k get pods --show-labels
k describe svc <name>       # check Selector field
\end{cheatcodebash}
\begin{cheatcodebash}
# DNS test:
k run tmp --image=busybox --rm -it --restart=Never -- nslookup <svc>.<ns>
\end{cheatcodebash}
\begin{cheatcodebash}
# HTTP test:
k run tmp --image=busybox --rm -it --restart=Never -- wget -qO- http://<svc>.<ns>:80
\end{cheatcodebash}
\begin{cheatcodebash}
# Check network policy interference:
k get netpol -A
\end{cheatcodebash}

\end{minipage}\hfill

% === Persistent Volume Claim Debugging
\tcbsubtitle{Persistent Volume Claim Debugging}
\begin{cheatcodebash}
k get pv    # check RECLAIM POLICY, STATUS
# Pending reasons: capacity mismatch, accessMode mismatch, storageClass mismatch
\end{cheatcodebash}

\begin{cheatcodebash}
### Exam helpers:
k explain persistentvolumeclaim.spec --recursive
k explain persistentvolume.spec.accessModes
# docs: "persistent volumes"
\end{cheatcodebash}

\begin{cheatcodebashbold}
### Inspect / debug
k describe pod <name>              # check Events!
k logs <pod> [-c <container>] [-p] # check logs [previous] 
\end{cheatcodebashbold}


\begin{cheatcodebashbold}
### tunnel from port 8080 on localhost to port 80 of pod
### CKA: verify pod's web UI/endpoint is working
k port-forward pod/<name> 8080:80
\end{cheatcodebashbold}



\begin{cheatcodebash}
### Files / events / lifecycle
k cp <pod>:/path /local
k get events --sort-by='.lastTimestamp' -A
k delete pod <name> $now       # force delete
\end{cheatcodebash}
