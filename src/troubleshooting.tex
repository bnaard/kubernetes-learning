        % === Pod Debugging ========================================================
        \tcbsubtitle{Pod Debugging}
        \begin{cheatcodebash}
        k get pods -A -o wide
        k describe pod <name> -n <ns>   # check Events!
        k logs <pod> [-c <container>]
        k logs <pod> --previous         # or -p crashed container
        k exec -it <pod> -- sh
        k exec -it <pod> -c <c> -- bash
        \end{cheatcodebash}
        \begin{cheatcodebash}
        # Ephemeral debug container (K8s 1.23+):
        # Shares process namespace with target container
        k debug -it <pod> --image=busybox --target=<container>
        # Copy pod for debugging (helps with CrashLoopBackOff:
        # copied pod can override entrypoint to keep it running)
        k debug <pod> -it --copy-to=<pod>-debug --image=busybox
        \end{cheatcodebash}
        \begin{cheatcodebashbold}
        ### tunnel from port 8080 on localhost to port 80 of pod
        ### verify pod's web UI/endpoint is working
        k port-forward pod/<name> 8080:80
        \end{cheatcodebashbold}
        \begin{cheatcodebash}
        ### Files / events / lifecycle
        k cp <pod>:/path /local
        k get events --sort-by='.lastTimestamp' -A
        k events --for pod/<name>      # events for resource
        k events -A --types=Warning    # cluster-wide warnings
        k delete pod <name> $now       # force delete
        \end{cheatcodebash}

        % === Node Debugging ========================================================
        \tcbsubtitle{Fix a Broken Node}
        {\fontsize{6.5}{7.5}\selectfont\sffamily
        \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
          \item \texttt{k get nodes -o wide} — identify \texttt{NotReady} node
          \item \texttt{k describe node <name>} — check Conditions, Events
          \item \texttt{ssh <node>}
          \item \texttt{systemctl status kubelet} (and \texttt{containerd})
          \item \texttt{journalctl -u kubelet | grep -i error} — find error
          \item \texttt{systemctl cat kubelet} — see config path, flags
          \item Fix config: wrong \texttt{--config} path, bad cert path, wrong \texttt{staticPodPath}, wrong kubelet binary in \texttt{10-kubeadm.conf}
          \item \texttt{systemctl daemon-reload} — after editing unit files!
          \item \texttt{systemctl restart kubelet}
          \item \texttt{systemctl enable kubelet} — ensure it survives reboot
          \item Verify: \texttt{k get nodes} — node becomes \texttt{Ready}
        \end{enumerate}}
        \begin{cheatcodebash}
k cordon <node>             # mark unschedulable
k drain <node> --ignore-daemonsets --delete-emptydir-data
k uncordon <node>
        \end{cheatcodebash}

      \tcbitem
        % === Control Plane Debug ==================================================
        \tcbsubtitle{Fix a Broken Control Plane}
        {\fontsize{6.5}{7.5}\selectfont\sffamily
        \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
          \item \texttt{k get pods -n kube-system} — or \texttt{crictl ps} if API is down
          \item \texttt{k logs kube-apiserver-<node> -n kube-system} — or \texttt{crictl logs <id>}
          \item Check static pod manifests in \texttt{/etc/kubernetes/manifests/}
          \item Common fixes: wrong image tag, bad volume mount path, typo in \texttt{--etcd-servers}, wrong cert path
          \item Save manifest — kubelet auto-restarts the pod
          \item Verify: \texttt{k get pods -n kube-system}
        \end{enumerate}}
        \begin{cheatcodebash}
# If API server unreachable:
crictl ps                   # list containers on node
crictl logs <container-id>
crictl inspect <container-id> # full state: config,
# image, mounts, pid, exit code, OOM kill, etc.
export KUBECONFIG=/etc/kubernetes/admin.conf
        \end{cheatcodebash}

          \tcbsubtitle{Troubleshoot a Service}
          {\fontsize{6.5}{7.5}\selectfont\sffamily
          \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
            \item \texttt{k get endpoints <svc>} — empty = selector mismatch
            \item Compare \texttt{svc} selector vs pod labels
            \item Check \texttt{targetPort} matches container port
            \item \texttt{k run tmp --rm -it --image=busybox -- wget -qO- <svc>:<port>}
            \item Check NetworkPolicy: \texttt{k get netpol -A}
          \end{enumerate}}

      \tcbitem
          \tcbsubtitle{Fix admin.conf / Kubeconfig}
          {\fontsize{6.5}{7.5}\selectfont\sffamily
          \begin{enumerate}[leftmargin=1.2em, itemsep=0pt, parsep=0pt, topsep=1pt]
            \item Symptom: \texttt{kubectl} cannot reach API server
            \item Check \texttt{/etc/kubernetes/admin.conf} (or \texttt{\$KUBECONFIG})
            \item Verify \texttt{server:} URL — must be \texttt{https://<control-plane-ip>:6443}
            \item Common issues: wrong port (e.g.\ \texttt{8443} instead of \texttt{6443}), wrong IP, \texttt{http} instead of \texttt{https}
            \item Fix and verify: \texttt{k get nodes}
          \end{enumerate}}
