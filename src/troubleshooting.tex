        % === Pod Debugging ========================================================
        \tcbsubtitle{Pod Debugging}
        \begin{cheatcodebash}
        k get pods -A -o wide
        k describe pod <name> -n <ns>   # check Events!
        k logs <pod> [-c <container>]
        k logs <pod> --previous         # or -p crashed container
        k exec -it <pod> -- sh
        k exec -it <pod> -c <c> -- bash
        \end{cheatcodebash}
        \begin{cheatcodebash}
        # Ephemeral debug container (K8s 1.23+):
        # Shares process namespace with target container
        k debug -it <pod> --image=busybox --target=<container>
        # Copy pod for debugging (helps with CrashLoopBackOff:
        # copied pod can override entrypoint to keep it running)
        k debug <pod> -it --copy-to=<pod>-debug --image=busybox
        \end{cheatcodebash}
        \begin{cheatcodebashbold}
        ### tunnel from port 8080 on localhost to port 80 of pod
        ### verify pod's web UI/endpoint is working
        k port-forward pod/<name> 8080:80
        \end{cheatcodebashbold}
        \begin{cheatcodebash}
        ### Files / events / lifecycle
        k cp <pod>:/path /local
        k get events --sort-by='.lastTimestamp' -A
        k delete pod <name> $now       # force delete
        \end{cheatcodebash}

        % === Node Debugging ========================================================
        \tcbsubtitle{Node Debugging}
        \begin{cheatcodebash}
        k get nodes -o wide
        k describe node <name>      # Conditions, Events
        k cordon <node>             # mark unschedulable
        k drain <node> --ignore-daemonsets --delete-emptydir-data
        k uncordon <node>
        \end{cheatcodebash}
        \begin{cheatcodebash}
        # On the node (SSH):
        systemctl status kubelet
        systemctl restart kubelet
        journalctl -u kubelet -f
        journalctl -u kubelet | grep -i error
        systemctl status containerd
        systemctl cat kubelet       # see config path
        # Usually: /var/lib/kubelet/config.yaml
        \end{cheatcodebash}

          % === Control Plane Debug ==================================================
          \tcbsubtitle{Control Plane Debug}
          \begin{cheatcodebash}
          # Static pods in kube-system:
          k get pods -n kube-system
          k logs kube-apiserver-<node> -n kube-system
          k logs etcd-<node> -n kube-system
          \end{cheatcodebash}

          \begin{cheatcodebash}
          # Edit control plane component:
          vi /etc/kubernetes/manifests/kube-apiserver.yaml
          \end{cheatcodebash}
          \begin{cheatcodebash}
          # If API server unreachable:
          crictl ps                   # list containers on node
          crictl logs <container-id>
          \end{cheatcodebash}
          \begin{cheatcodebash}
          # Kubeconfig:
          export KUBECONFIG=/etc/kubernetes/admin.conf
          \end{cheatcodebash}

        \tcbitem
          % === Service Debugging =====================================================
          \tcbsubtitle{Service \& Networking Debugging}
          \begin{cheatcodebash}
          # Check if svc routes to pods:
          k get endpoints <svc>       # empty = label mismatch
          k get pods --show-labels
          k describe svc <name>       # check Selector field
          \end{cheatcodebash}
          \begin{cheatcodebash}
          # DNS test:
          k run tmp --image=busybox --rm -it --restart=Never -- nslookup <svc>.<ns>
          \end{cheatcodebash}
          \begin{cheatcodebash}
          # HTTP test:
          k run tmp --image=busybox --rm -it --restart=Never -- wget -qO- http://<svc>.<ns>:80
          \end{cheatcodebash}
          \begin{cheatcodebash}
          # Check network policy interference:
          k get netpol -A
          \end{cheatcodebash}

        \tcbitem
          % === Certificate Debugging ================================================
          \tcbsubtitle{Certificate Debugging}
          \begin{cheatcodebash}
          # Check cert expiry:
          kubeadm certs check-expiration
          # Inspect a specific cert:
          openssl x509 -in /etc/kubernetes/pki/apiserver.crt \
            -noout -text -dates
          # Verify cert matches key:
          openssl x509 -in cert.crt -noout -modulus | md5sum
          openssl rsa  -in cert.key -noout -modulus | md5sum
          \end{cheatcodebash}
          \begin{cheatcodebash}
          # Renew all certs:
          kubeadm certs renew all
          # Then restart control plane (kubelet picks up new certs):
          systemctl restart kubelet
          \end{cheatcodebash}
          \begin{cheatcodebash}
          # Common symptoms:
          # - "x509: certificate has expired" in kubelet/apiserver logs
          # - "Unable to connect to the server" from kubectl
          # docs: search "manage tls certificates"
          \end{cheatcodebash}
