% === Pod Creation ==========================================================
\tcbsubtitle{Imperative}
\begin{cheatcodebashbold}
### Basic pod
k run <name> --image=<img> --restart=Never
k run busybox --image=busybox -- sleep 3600
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
### Temporary interactive shell
k run tmp --image=busybox --rm -it [--restart=Never] -- sh
\end{cheatcodebashbold}

\begin{cheatcodebash}
### With port, env var and label
k run <name> --image=<img> --port=80 --env="K=v" -l "a=b"
\end{cheatcodebash}

\begin{cheatcodebashbold}
### Get pod spec, tee prints on tty and into file
k run <name> --image=<img> $do | tee pod.yaml
\end{cheatcodebashbold}

\begin{cheatcodebashbold}
# Force delete pod
k delete pod <name> $now
\end{cheatcodebashbold}

% === Pod Spec Key Fields ===================================================
\tcbsubtitle{Spec Essentials}
\begin{cheatcodebashbold}
spec:                            # <pod>.spec
  restartPolicy: Always|Never|OnFailure
  serviceAccountName: <sa>
  hostNetwork: true              # use node network
  initContainers:
  - name: init
    image: busybox
    command: ['sh','-c','echo init']
  containers:
  - name: app
    image: nginx:1.21
    ports: [{containerPort: 80}]
    command: ["/bin/sh"]
    args: ["-c", "while true; do sleep 3; done"]
    env:
    - name: MY_VAR
      value: "literal"
    resources:
      requests: {cpu: 100m, memory: 128Mi}
      limits:   {cpu: 500m, memory: 256Mi}
    volumeMounts:
    - {name: vol, mountPath: /data}
  volumes:
  - name: vol
    persistentVolumeClaim:
      claimName: my-pvc
\end{cheatcodebashbold}
