  \tcbsubtitle{What is a CRD}
  {\fontsize{6.5}{7.5}\selectfont\sffamily
  A \textbf{Custom Resource Definition} extends the Kubernetes API with
  user-defined resource types. Once registered, custom resources (CRs) are
  managed like built-in resources (\texttt{get}, \texttt{describe},
  \texttt{delete}, etc.).\par}

  \tcbsubtitle{CRD Spec Structure}
  \begin{cheatcodeyaml}
apiVersion: apiextensions.k8s.io/v1
kind: CustomResourceDefinition
metadata:
  name: backups.stable.example.com
spec:
  group: stable.example.com
  scope: Namespaced              # or Cluster
  names:
    kind: Backup
    plural: backups
    singular: backup
    shortNames: [bk]
  versions:
  - name: v1
    served: true
    storage: true
    schema:
      openAPIV3Schema:
        type: object
        properties:
          spec:
            type: object
            properties:
              schedule:
                type: string
              retainDays:
                type: integer
  \end{cheatcodeyaml}

  \tcbsubtitle{Creating \& Using Custom Resources}
  \begin{cheatcodebash}
k apply -f crd.yaml            # register the CRD
k get backups                  # list custom resources
k describe backup mybackup     # inspect a CR instance
  \end{cheatcodebash}

  \cmdhead{Minimal CR Instance}
  \begin{cheatcodeyaml}
apiVersion: stable.example.com/v1
kind: Backup
metadata:
  name: mybackup
spec:
  schedule: "0 2 * * *"
  retainDays: 7
  \end{cheatcodeyaml}

  \tcbsubtitle{Operator Pattern}
  {\fontsize{6.5}{7.5}\selectfont\sffamily
  \textbf{CRD + custom controller = Operator.}
  The controller watches CR events (create/update/delete) and reconciles
  actual state to match the desired spec.\par}

  \tcbsubtitle{Discovering \& Installing Operators}
  \begin{cheatcodebash}
k get crd                       # list all CRDs
k explain <crd-kind>            # schema of a custom resource
k describe crd <name>           # versions, scope, schema
  \end{cheatcodebash}
  \begin{cheatcodebash}
# Install via Helm chart:
helm install my-operator <chart-repo>/<chart-name>
# Install via OLM (if available):
k get operators                 # list installed operators
k get csv -A                    # ClusterServiceVersions
  \end{cheatcodebash}
