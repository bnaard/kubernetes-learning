    % === ServiceAccount =======================================================
    \tcbsubtitle{ServiceAccount}
    \cmdhead{Imperative}
    \begin{cheatcodebash}
    k create sa <name> -n <ns>
    k get sa -n <ns>
    \end{cheatcodebash}

    \cmdhead{Spec Essentials: Assign to Pod}
    \begin{cheatcodebash}
    spec:                  # <pod>.spec
      serviceAccountName: <name>
      automountServiceAccountToken: false
    \end{cheatcodebash}

    \cmdhead{Tokens}
    \begin{cheatcodebash}
    # Create token (K8s 1.24+) → manual, prints token to terminal, admin/CI use
    k create token <sa-name> --duration=1h
    # Token mounted at: → automatically by kubelet for pods with a service account
    # /var/run/secrets/kubernetes.io/serviceaccount/
    \end{cheatcodebash}

    \cmdhead{Create role and bind it}
    \begin{cheatcodebashbold}
    k create role r --verb=get,list --resource=pods
    k create rolebinding rb --role=r --serviceaccount=<ns>:<sa>
    \end{cheatcodebashbold}

    \cmdhead{Help}
    \begin{cheatcodebash}
    k explain serviceaccount
    k explain pod.spec.serviceAccountName
    \end{cheatcodebash}


    \cmdhead{\url{https://kubernetes.io/docs/}}
    \begin{cheatcodebash}
    search: "service accounts"
    \end{cheatcodebash}

    % === SecurityContext ======================================================
    \tcbsubtitle{SecurityContext}

    \begin{tcbitemize}[
      raster columns=2, raster equal height=rows,
      blanker, cheat subtitle
]

      \tcbitem
        \cmdhead{Spec: Pod level (applies to all containers)}
        \begin{cheatcodebash}
        spec:                 # <pod>.spec
          securityContext:
            runAsUser: 1000
            runAsGroup: 3000
            fsGroup: 2000     # volume ownership
            runAsNonRoot: true
            seccompProfile:
              type: RuntimeDefault
        \end{cheatcodebash}

        \cmdhead{Spec: Container level (overrides pod level)}
        \begin{cheatcodebash}
          containers:         # <pod>.spec.containers
          - securityContext:
              allowPrivilegeEscalation: false
              readOnlyRootFilesystem: true
              privileged: false
              capabilities:
                add: ["NET_ADMIN", "SYS_TIME"]
                drop: ["ALL"]
        \end{cheatcodebash}


        \cmdhead{Help}
        \begin{cheatcodebash}
        k explain pod.spec.securityContext --recursive
        k explain pod.spec.containers.securityContext --recursive
        \end{cheatcodebash}

        \cmdhead{\url{https://kubernetes.io/docs/}}
        \begin{cheatcodebash}
        search: "security context"
        \end{cheatcodebash}

      \tcbitem
        \cmdhead{Linux Capabilities}
        \begin{tcolorbox}

          Capability & Purpose \\
          \texttt{NET\_ADMIN} & Configure interfaces, routing, firewall rules \\
          \texttt{NET\_RAW} & Use raw/packet sockets (ping, tcpdump) \\
          \texttt{NET\_BIND\_SERVICE} & Bind to ports below 1024 \\
          \texttt{SYS\_TIME} & Set system clock \\
          \texttt{SYS\_PTRACE} & Trace/debug processes (strace, gdb) \\
          \texttt{SYS\_ADMIN} & Broad: mount, namespaces, syslog — avoid! \\
          \texttt{CHOWN} & Change file ownership \\
          \texttt{DAC\_OVERRIDE} & Bypass file read/write/exec permission checks \\
          \texttt{SETUID / SETGID} & Change process UID/GID \\
          \texttt{KILL} & Send signals to other processes \\
          \texttt{AUDIT\_WRITE} & Write to kernel audit log \\
            
        \end{tcolorbox}
        \note{Best practice: \texttt{drop: ["ALL"]} then add back only what's needed.}
