% === RBAC Overview ========================================================
\tcbsubtitle{RBAC}
{\small\sffamily
\begin{tblr}{
  colspec={lX},
  hline{2}={0.4pt},
  rows={rowsep=0.5pt},
  row{1}={font=\bfseries},
}
Resource & Scope \\
Role & namespaced permissions \\
ClusterRole & cluster-wide permissions \\
RoleBinding & binds Role/ClusterRole in ns \\
ClusterRoleBinding & binds ClusterRole cluster-wide \\
\end{tblr}}
\begin{cheatcodebash}
# Create Role (namespaced):
k create role <name> \
  --verb=get,list,watch \
  --resource=pods,deployments -n <ns>
# Create ClusterRole:
k create clusterrole <name> \
  --verb='*' --resource='*'
# Bind to ServiceAccount:
k create rolebinding <name> \
  --role=<role> \
  --serviceaccount=<ns>:<sa> -n <ns>
k create clusterrolebinding <name> \
  --clusterrole=<cr> --user=<user>
# Verify:
k auth can-i get pods --as=<user>
k auth can-i get pods \
  --as=system:serviceaccount:<ns>:<sa>
k auth can-i --list -n <ns>
\end{cheatcodebash}

Role YAML:
\begin{cheatcodebash}
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata: {name: pod-reader, namespace: <ns>}
rules:
- apiGroups: [""]           # "" = core API group
  resources: ["pods", "pods/log"]
  verbs: ["get","list","watch"]
- apiGroups: ["apps"]
  resources: ["deployments"]
  verbs: ["get","list","create","update"]
\end{cheatcodebash}

% === ServiceAccount =======================================================
\tcbsubtitle{ServiceAccount}
\begin{cheatcodebash}
k create sa <name> -n <ns>
k get sa -n <ns>
# Assign to Pod:
spec:
  serviceAccountName: <name>
  automountServiceAccountToken: false
# Create token (K8s 1.24+):
k create token <sa-name> --duration=1h
# Token mounted at:
# /var/run/secrets/kubernetes.io/serviceaccount/
\end{cheatcodebash}

% === SecurityContext ======================================================
\tcbsubtitle{SecurityContext}
\begin{cheatcodebash}
# Pod level (applies to all containers):
spec:
  securityContext:
    runAsUser: 1000
    runAsGroup: 3000
    fsGroup: 2000          # volume ownership
    runAsNonRoot: true
    seccompProfile:
      type: RuntimeDefault
# Container level (overrides pod level):
  containers:
  - securityContext:
      allowPrivilegeEscalation: false
      readOnlyRootFilesystem: true
      privileged: false
      capabilities:
        add: ["NET_ADMIN", "SYS_TIME"]
        drop: ["ALL"]
\end{cheatcodebash}
